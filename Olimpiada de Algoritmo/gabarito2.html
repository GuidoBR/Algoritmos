function PreencherVaga ( $Vagas, $Fileira, $Lado, $Poltrona ) {



	if ( $Lado == "direito" ) {



		switch ( $Poltrona ) {

			case "janela":   $i = 1;

			case "corredor": $i = 2;

		}



	}



	if ( $Lado == "esquerdo" ) {



		switch ( $Poltrona ) {

			case "janela":   $i = 4;

			case "corredor": $i = 3;

		}



	}



	if ( $Vagas[$i][$Fileira] == "0" ) {

  	$Vagas[$i][$Fileira] = "x";

	} else {

		print("A vaga já estava ocupada");

	}



}

function FileirasVazias ( $Vagas ) {



	for( $i = 1; $i <= 9; $i++ ) {



		$Vazia = true;



		for( $j = 1; $j <= 4; $j++ ) {



			if ( $Vagas[$j][$i] == "x" ) {

				$Vazia = false;

			}



		}



		if ( $Vazia == true ) {

			print($i);

		}



	}



}

function Relatorio ( $Vagas ) {



  for ( $i = 1; $i <= 9; $i++ ) {

    print($i . " - ");



      for ( $j = 1; $j <= 4; $j++ ) {

        print($Vagas[$j][$i]);



        if ( $Vagas[$j][$i] == '0' ) {

          $Vazia = $Vazia + 1;

        } else {

          if ( $j == 1 || $j == 4 ) {

            $Janela = $Janela + 1;

          } else {

            $Corredor = $Corredor + 1;

          }

        }



      }



    print("\n");

  }



  print($Janela . " Pessoas Janela ");

  print($Corredor . " Pessoas Corredor ");

  print($Janela + $Corredor . " Total Ocupadas ");

  print($Vazio . " Total de Vazias ");



}

function TotalArrecadado ( $CidadesEnt, $CidadesSai ) {



  // Cidade A até Cidade B

  $Pessoas = $CidadesEnt[1];

  $Total = $Pessoas * 15;



  //Cidade B até Cidade C, C até D e D até E

  for( $i = 2; $i <= 4; $i++ ) {



    $Pessoas = ( $Pessoas + $CidadesEnt[$i] ) - $CidadesSai[i];



	switch ( $i ) {

		case '2': $Total = $Total + $Pessoas * 12;

		case '3': $Total = $Total + $Pessoas * 35;

		case '4': $Total = $Total + $Pessoas * 53;

	}

  }



  print("Total arrecadado na rota: " . $Total);

}
